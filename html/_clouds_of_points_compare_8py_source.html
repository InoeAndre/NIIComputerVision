129 def InvPose(Pose):
130  '''Compute the inverse transform of Pose'''
131  PoseInv = np.zeros(Pose.shape,Pose.dtype)
132  PoseInv[0:3,0:3] = LA.inv(Pose[0:3,0:3])
133  PoseInv[0:3,3] = -np.dot(PoseInv[0:3,0:3],Pose[0:3,3])
134  PoseInv[3,3] = 1.0
135  return PoseInv
