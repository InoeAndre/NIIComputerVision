53  def findSlope(self,A,B):
54  '''Get the slope of a line made from two point A and B or the distance in one axes'''
55  A = A.astype(np.float32)
56  B = B.astype(np.float32)
57  diffY = B[1]-A[1]
58  diffX = A[0]-B[0]
59  dist = np.sqrt(np.square(diffY) + np.square(diffX))
60  a = diffY/dist # normalized distance
61  b = diffX/dist # normalized distance
62  c = -a*A[0]-b*A[1]
63  return np.array([a,b,c])

154  def polygon(self,slopes,ref, limit ):
155  ''' This function test the sign of alpha = (a[k]*j+b[k]*i+c[k])*ref[k]
156  to know whether a point is within a polygon or not'''
157  start_time = time.time()
158  line = self.depthImage.shape[0]
159  col = self.depthImage.shape[1]
160  res = np.zeros([line,col],np.bool)
161  alpha = np.zeros([1,limit])
162  for i in range(line):
163  for j in range(col):
164  for k in range(limit):
165  alpha[0][k] = (slopes[0][k]*j+slopes[1][k]*i+slopes[2][k])*ref[0,k]
166  alpha_positif = (alpha >= 0)
167  if alpha_positif.all():
168  res[i,j]=True
169  elapsed_time = time.time() - start_time
170  print "polygon: %f" % (elapsed_time)
171  return res

173  def polygon_optimize(self,slopes,ref, limit ):
174  ''' This function test the sign of alpha = (a[k]*j+b[k]*i+c[k])*ref[k]
175  to know whether a point is within a polygon or not'''
176  #start_time = time.time()
177  line = self.depthImage.shape[0]
178  col = self.depthImage.shape[1]
179  res = np.ones([line,col])
180 
181  #create a matrix containing in each pixel its indices
182  lineIdx = np.array([np.arange(line) for _ in range(col)]).reshape(col,line).transpose()
183  colIdx = np.array([np.arange(col) for _ in range(line)]).reshape(line,col)
184  depthIdx = np.ones([line,col])
185  ind = np.stack( (colIdx,lineIdx,depthIdx), axis = 2)
186 
187  alpha = np.zeros([line,col,limit])
188  alpha= np.dot(ind,slopes)
189  # for each k (line) if the points (ref and the current point in alpha) are on the same side then the operation is positiv
190  for k in range(limit):
191  alpha[:,:,k]=( (np.dot(alpha[:,:,k],ref[0][k])) >= 0)
192  # make sure that each point are on the same side as the reference for all line of the polygon
193  for k in range(limit):
194  res = res*alpha[:,:,k ]
195  res = (res>0)
196  #elapsed_time = time.time() - start_time
197  #print "polygon_optimize: %f" % (elapsed_time)
198  return res

200  def polygonOutline(self,points):
201  '''Find a polygon on the image through the points given in points
202  /arg points : array of points which are the corners of the polygon to find'''
203  line = self.depthImage.shape[0]
204  col = self.depthImage.shape[1]
205  im_out = np.zeros([line,col],np.uint8)
206  points = points.astype(np.float64)
207  n = points.shape[0]
208  i = 2
209  d = 0
210  # trace the segment
211  ptB = np.zeros(points.shape)
212  ptB[-1]=points[0]
213  for i in range(0,points.shape[0]-1):
214  ptB[i] = points[i+1]
215 
216  M = np.zeros([line,col],np.uint8)
217  for i in range(n-d):
218  A = points[i,:]
219  B = ptB[i,:]
220  slopes = self.findSlope(A,B)
221  if np.abs(slopes[0]) > np.abs(slopes[1]):
222  if A[1] > B[1]:
223  tmp = B
224  B = A
225  A = tmp
226  for y in range(int(A[1]),int(B[1])+1):
227  x = np.round(-(slopes[1]*y+slopes[2])/slopes[0])
228  M[int(y),int(x)]= 1
229  else :
230  if A[0] > B[0]:
231  tmp = B
232  B = A
233  A = tmp
234  for x in range(int(A[0]),int(B[0])+1):
235  y = np.round(-(slopes[0]*x+slopes[2])/slopes[1])
236  M[int(y),int(x)]= 1
237 
239  im_floodfill = M.copy()
240  im_floodfill = im_floodfill.astype(np.uint8)
241 
242  # Mask used to flood filling.
243  # Notice the size needs to be 2 pixels than the image.
244  h, w = M.shape[:2]
245  mask = np.zeros((h+2, w+2), np.uint8)
246 
247  # Floodfill from point (0, 0)
248  cv2.floodFill(im_floodfill, mask, (0,0), 255)
249 
250  # Invert floodfilled image
251  im_floodfill_inv = cv2.bitwise_not(im_floodfill)
252 
253  # Combine the two images to get the foreground.
254  im_out = M | im_floodfill_inv
255  return im_out
