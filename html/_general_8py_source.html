35 :param Pose: 4*4 Matrix of the camera pose

40 def InvPose(Pose):
41  PoseInv = np.zeros(Pose.shape, Pose.dtype)
42  # Inverse rotation part R^(-1)
43  PoseInv[0:3, 0:3] = LA.inv(Pose[0:3, 0:3])
44  # Inverse Translation part R^(-1)*T
45  PoseInv[0:3, 3] = -np.dot(PoseInv[0:3, 0:3], Pose[0:3, 3])

49 def normalized_cross_prod(a, b):
50  '''
51  compute the cross product of 2 vectors and normalized it
52  :param a: first 3 elements vector
53  :param b: second 3 elements vector
54  :return: the normalized cross product between 2 vector
55  '''
56  res = np.zeros(3, dtype="float")
57  if (LA.norm(a) == 0.0 or LA.norm(b) == 0.0):
58  return res
59  a = a / LA.norm(a)
60  b = b / LA.norm(b)
61  res[0] = a[1] * b[2] - a[2] * b[1]
62  res[1] = -a[0] * b[2] + a[2] * b[0]
63  res[2] = a[0] * b[1] - a[1] * b[0]
64  if (LA.norm(res) > 0.0):
65  res = res / LA.norm(res)
66  return res

69 def division_by_norm(mat, norm):
70  '''
71  This fonction divide a n by m by p=3 matrix, point by point, by the norm made through the p dimension
72  It ignores division that makes infinite values or overflow to replace it by the former mat values or by 0
73  :param mat:
74  :param norm:
75  :return:
76  '''
77  for i in range(3):
78  with np.errstate(divide='ignore', invalid='ignore'):
79  mat[:, :, i] = np.true_divide(mat[:, :, i], norm)
